

CREATE DATABASE ementor;


use ementor;


CREATE TABLE ementor.Pessoas (nome VARCHAR(20) UNIQUE, dataNascimento VARCHAR(10), CPF VARCHAR(15) PRIMARY KEY, telefone VARCHAR(15));


CREATE TABLE ementor.Usuarios (nome VARCHAR(20) PRIMARY KEY, senha VARCHAR(20), FOREIGN KEY(nome) REFERENCES ementor.Pessoas(nome));
